  fsw/src/cf_cfdp.c
  fsw/src/cf_cfdp_r.c
  fsw/src/cf_cfdp_s.c
  fsw/src/cf_cfdp_sbintf.c
  fsw/src/cf_cfdp_dispatch.c
  fsw/src/cf_chunk.c

  fsw/src/cf_codec.c

)

  list(APPEND APP_SRC_FILES fsw/src/cf_eds_dispatch.c)
else()
